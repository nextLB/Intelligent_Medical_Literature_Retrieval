## 二、核心算法实现

### 2.1 文本分类算法

#### 2.1.1 BERT分类模型架构
```python
class BertClassifier(nn.Module):
    def __init__(self, num_classes=7, model_name='bert-base-chinese', dropout=0.3):
        super(BertClassifier, self).__init__()
        self.bert = BertModel.from_pretrained(model_name)
        self.dropout = nn.Dropout(dropout)
        self.classifier = nn.Linear(self.bert.config.hidden_size, num_classes)
    
    def forward(self, input_ids, attention_mask):
        outputs = self.bert(input_ids=input_ids, attention_mask=attention_mask)
        pooled_output = outputs.pooler_output
        pooled_output = self.dropout(pooled_output)
        logits = self.classifier(pooled_output)
        return logits
```

#### 2.1.2 训练流程
1. **数据加载**：从CSV文件加载训练数据
2. **文本编码**：使用BERT分词器将文本转换为token IDs
3. **模型配置**：
   - 优化器：AdamW（学习率2e-5）
   - 损失函数：交叉熵损失
   - 学习率调度：线性warmup策略
4. **训练循环**：
   - 每个epoch包含训练和验证阶段
   - 梯度裁剪防止梯度爆炸
   - 保存最佳验证准确率的模型

#### 2.1.3 预测流程
1. **文本预处理**：清理和截断文本
2. **编码输入**：使用相同的分词器处理
3. **模型推理**：前向传播获取logits
4. **结果解析**：Softmax归一化得到类别概率

### 2.2 文本向量化算法

#### 2.2.1 TextVectorizer实现
```python
class TextVectorizer:
    def __init__(self, model_name='bert-base-chinese'):
        self.device = torch.device('cuda' if torch.cuda.is_available() else 'cpu')
        self.tokenizer = BertTokenizer.from_pretrained(model_name)
        self.model = BertModel.from_pretrained(model_name)
        self.model = self.model.to(self.device)
        self.model.eval()
        self.max_length = 256
    
    def get_vector(self, text):
        inputs = self.tokenizer(
            text, add_special_tokens=True, max_length=self.max_length,
            padding='max_length', truncation=True, return_tensors='pt'
        )
        input_ids = inputs['input_ids'].to(self.device)
        attention_mask = inputs['attention_mask'].to(self.device)
        
        with torch.no_grad():
            outputs = self.model(input_ids=input_ids, attention_mask=attention_mask)
            vector = outputs.last_hidden_state[:, 0, :].cpu().numpy()[0]
        
        return vector
```

#### 2.2.2 向量存储与检索
- 使用内存字典存储文献ID到向量的映射
- 支持动态添加和删除文献向量
- 向量归一化处理保证相似度计算准确

### 2.3 相似度搜索算法

#### 2.3.1 SimilaritySearch实现
```python
class SimilaritySearch:
    def __init__(self):
        self.vectorizer = get_text_vectorizer()
        self.literature_vectors = {}
        self.literature_texts = {}
    
    def search_similar(self, query, top_k=10, literature_ids=None):
        query_vector = self.vectorizer.get_vector(query)
        candidate_ids = literature_ids if literature_ids else list(self.literature_vectors.keys())
        
        if not candidate_ids:
            return []
        
        vectors = []
        for lid in candidate_ids:
            if lid in self.literature_vectors:
                vectors.append(self.literature_vectors[lid])
        
        vectors = np.array(vectors)
        similarities = cosine_similarity([query_vector], vectors)[0]
        
        results = []
        for i, lid in enumerate(candidate_ids):
            results.append({
                'literature_id': lid,
                'similarity': float(similarities[i])
            })
        
        results.sort(key=lambda x: x['similarity'], reverse=True)
        return results[:top_k]
```

#### 2.3.2 相似度计算原理
- 使用余弦相似度：cosθ = (A·B)/(||A||·||B||)
- BERT输出的768维向量
- 相似度范围：[-1, 1]，实际值通常为[0, 1]

### 2.4 关键词检索算法

#### 2.4.1 分词与匹配

```python
def keyword_search_tokens(query):
    """从用户输入拆出可用于匹配的关键词"""
    q = (query or '').strip()
    if not q:
        return []
    tokens = []
    if len(q) >= 2:
        tokens.append(q)
    for w in jieba.cut_for_search(q):
        w = w.strip()
        if len(w) >= 2:
            tokens.append(w)
    seen = set()
    out = []
    for t in tokens:
        if t not in seen:
            seen.add(t)
            out.append(t)
    return out
```

#### 2.4.2 多字段匹配
- 标题字段：title__icontains
- 摘要字段：abstract__icontains
- 关键词字段：keywords__icontains
- 使用Q对象构建OR查询条件

## 四、业务逻辑实现

### 4.1 检索功能实现

#### 4.1.1 关键词检索流程
```python
def keyword_search(request):
    # 1. 解析请求参数
    query = request.GET.get('q', '').strip()
    
    # 2. 分词处理
    tokens = keyword_search_tokens(query)
    
    # 3. 构建查询条件
    combined = Q()
    for t in tokens:
        combined |= Q(title__icontains=t) | Q(abstract__icontains=t) | Q(keywords__icontains=t)
    
    # 4. 执行查询
    results = Literature.objects.filter(combined)
    
    # 5. 结果格式化
    formatted_results = []
    for lit in results:
        formatted_results.append({
            'id': lit.id,
            'title': lit.title,
            'abstract': lit.abstract[:200] + '...',
            'category': lit.category.name if lit.category else ''
        })
    
    # 6. 保存搜索历史
    SearchHistory.objects.create(query=query, search_type='keyword', results_count=len(results))
    
    return JsonResponse({'results': formatted_results})
```

#### 4.1.2 语义检索流程
```python
def semantic_search_view(request):
    # 1. 解析参数
    query = request.POST.get('query', '')
    
    # 2. 获取文本向量化器
    vectorizer = get_text_vectorizer()
    
    # 3. 获取查询向量
    query_vector = vectorizer.get_vector(query)
    
    # 4. 计算所有文献相似度
    similarities = []
    for lit in Literature.objects.all():
        text = f"{lit.title} {lit.abstract}"
        if not lit.text_vector:
            lit_vector = vectorizer.get_vector(text)
            lit.text_vector = json.dumps(lit_vector.tolist())
            lit.save()
        else:
            lit_vector = np.array(json.loads(lit.text_vector))
        
        similarity = cosine_similarity([query_vector], [lit_vector])[0][0]
        similarities.append((lit, similarity))
    
    # 5. 排序并返回结果
    similarities.sort(key=lambda x: x[1], reverse=True)
    results = []
    for lit, sim in similarities[:20]:
        results.append({
            'id': lit.id,
            'title': lit.title,
            'similarity': round(sim, 4)
        })
    
    return JsonResponse({'results': results})
```

#### 4.2.2 批量分类
```python
def auto_classify_all(request):
    # 获取未分类文献
    unclassified = Literature.objects.filter(category__isnull=True)
    
    classifier = get_bert_classifier()
    results = []
    
    for lit in unclassified:
        text = f"{lit.title} {lit.abstract}"
        pred = classifier.predict(text)
        
        # 查找对应的分类
        category = LiteratureCategory.objects.filter(name=pred['category_name']).first()
        if category:
            lit.category = category
            lit.save()
        
        results.append({
            'id': lit.id,
            'title': lit.title[:50],
            'category': pred['category_name'],
            'confidence': pred['confidence']
        })
    
    return JsonResponse({'results': results})
```

```python
def train_model_stream(request):
    # 解析训练参数
    num_epochs = int(request.POST.get('num_epochs', 3))

    def generate():

        for epoch in range(num_epochs):
            # ... 训练逻辑 ...
```
